Copia il codice Python qui sotto in un file main.py e crea il file requirements.txt con flask

from flask import Flask, request, jsonify, render_template_string
from datetime import datetime
import random

app = Flask(__name__)

users = {}

HTML = """
<html lang='it'>
<head><meta charset='UTF-8'/><title>FriendlyTouch</title></head>
<body><h1>Benvenuto su FriendlyTouch</h1><p>API attive!</p></body>
</html>"""

@app.route('/')
def home():
    return render_template_string(HTML)

@app.route('/register', methods=['POST'])
def register():
    data = request.get_json()
    username = data.get('username')
    latitude = data.get('latitude')
    longitude = data.get('longitude')
    timestamp = datetime.now().strftime('%Y-%m-%d %H:%M:%S')

    if not all([username, latitude, longitude]):
        return jsonify({'status': 'error', 'message': 'Dati mancanti'}), 400

    try:
        latitude = float(latitude)
        longitude = float(longitude)
    except:
        return jsonify({'status': 'error', 'message': 'Coordinate non valide'}), 400

    users[username] = {'latitude': latitude, 'longitude': longitude, 'joined': timestamp}
    return jsonify({'status': 'success', 'message': f'{username} registrato!'})

@app.route('/users', methods=['GET'])
def get_users():
    return jsonify({'status': 'success', 'users': users})

@app.route('/random_user', methods=['GET'])
def random_user():
    if not users:
        return jsonify({'status': 'error', 'message': 'Nessun utente'}), 404
    user = random.choice(list(users.items()))
    return jsonify({'status': 'success', 'username': user[0], 'data': user[1]})

if __name__ == '__main__':
    app.run(host='0.0.0.0', port=5000)
